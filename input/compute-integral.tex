\subsection{The Abelian Automorphisms}
\subsubsection{The Automorphism $\varphi_{r,k}(a)$}
Let $a\in\mathbb{Q}_p$, we define a map $\varphi_{r,k}(a):\sfrac{\mathcal{L}_{n,p}}{\gamma_2}\rightarrow\mathcal{L}_{n,p}$ in the following way:
\[
    \varphi_{r,k}(a)(e_i)=\begin{cases}
        e_{k,k+1}+a{e_{k,k+r}} & : i=k\\
        e_{k+r,k+r+1}-a{e_{k+1,k+r+1}} & : i=k+r\\
        e_{i,i+1} & : i\notin\{k,k+r\}
    \end{cases},
\]
for some $2\leq{r}\leq{n-2}$ and $0\leq{k}\leq{n-r}$.\\
Note that we include also the case $k=0$ where the index $i=k=0$ does not exist, and the case $k=n-r$ where the index $i=k+r=n$ does not exist.
\begin{proposition}
$\varphi_{r,k}(a)$ extends to an $\mathcal{L}_{n,p}$-automorphism which operates diagonally on $\gamma_2\mathcal{L}_{n,p}$.
\end{proposition}
\begin{proof}
We need to define the image $\varphi_{r,k}(a)(e_{ij})$ for any element $e_{ij}\in\mathcal{L}_{n,p}$, and show that it is compliant as an automorphism on $\mathcal{L}_{n,p}$. We do this by induction.\\
For $n=4$ we define the extension explicitly as follows:\\
(denote $\psi=\varphi_{21}(a)$ for simplicity)\\
$\psi(e_{13})=[\psi(e_1),\psi(e_2)]=[e_1+ae_{13},e_2]=e_{13}=-[e_2,e_1+ae_{13}]=-[\psi(e_2),\psi(e_1)]$.\\
$\psi(e_{24})=[\psi(e_2),\psi(e_3)]=[e_2,e_3-ae_{24}]=e_{24}=-[e_3-ae_{24},e_2]=-[\psi(e_3),\psi(e_2)]$.\\
$\psi(e_{14})=[\psi(e_1),\psi(e_{24})]=[e_1+ae_{13},e_{24}]=e_{14}=-[e_{24},e_1+ae_{13}]=-[\psi(e_3),\psi(e_{13})]$.\\
But we also have\\
$\psi(e_{14})=[\psi(e_{13}),\psi(e_3)]=[e_{13},e_3-ae_{24}]=e_{14}=-[e_3-ae_{24},e_{13}]=-[\psi(e_3),\psi(e_{13})]$,\\
which shows that the extended $\varphi_{21}(a)$ is well-defined and $\varphi_{21}(e_{ij})=e_{ij}$ for $(i,j)\in\{(1,3),(2,4),(1,4)\}$.\\
For $n>4$, we consider the injection $\iota:\mathcal{L}_{n-1,p}\rightarrow\mathcal{L}_{n,p}$, defined by abuse of notation as the identity $\iota(e_i)=e_i$ which implies that $\varphi_{r,k}(a)(\iota(e_i))=\varphi_{r,k}(a)(e_i)$, for all $1\leq{i}\leq{n-2}$, because $\varphi_{r,k}(a)$ does not depend on $n$, and therefore we have also $\varphi_{r,k}(a)(\iota(e_{i,j}))=\varphi_{r,k}(a)(e_{i,j})$, for all $1\leq{i}<j\leq{n-1}$.\\
Consider an element $e_{i,n}$, for some $1\leq{i}\leq{n-1}$. We define the image of this element by $\varphi_{r,k}(a)(e_{i,n})=\varphi_{r,k}(a)([e_{i,n-1},e_{n-1}])=[\varphi_{r,k}(a)(e_{i,n-1}),\varphi_{r,k}(a)(e_{n-1})]=\\=[\varphi_{r,k}(a)(\iota(e_{i,n-1})),\varphi_{r,k}(a)(e_{n-1})]=[e_{i,n-1},\varphi_{r,k}(a)(e_{n-1})]=[e_{i,n-1},e_{n-1}]=e_{i,n}=-[e_{n-1},e_{i,n-1}]=-[\varphi_{r,k}(a)(e_{n-1}),\varphi_{r,k}(a)(e_{i,n-1})]$.
\end{proof}
Thus, the structure of $\varphi_{r,k}$ as a block matrix $M$ is having the identity matrices $I_{n-1},I_{n-2},\dots,1$ on the main blocks diagonal, and the block $M_{1,r}$ in row $1$ and column $r$ of the blocks of $M$. The index $i$ is the row index of the block, which is also the row index of the whole matrix. The column index is $k$. which means that if we denote the matrix of $\varphi_{r,k}(a)$ by $N_{r,k}(a)$, then it forms as
\[N_{r,k}(a):=\begin{bmatrix}
    I_{n-1} &  & M_{1,r} & & \\
     & I_{n-2} &  & & \\
     & & \ddots &  & \\
          & & & I_2 & \\
               & & & & 1\\
\end{bmatrix},\]
where
\[
M_{1,r}=(m_{ij}):=\begin{bmatrix}
    & & &\\
    & a & &\\
    & & &\\
    & & -a &\\
    & & &\\
\end{bmatrix},
\]
that is, $m_{k,k}=a$ and $m_{k+r,k+1}=-a$.

\begin{proposition}
Fix $r$ and $k$, and let $N_{r,k}:=\{N_{r,k}(a) : a\in\mathbb{Q}_p\}$, then $N_{r,k}$ is an abelian subgroup of $G_n^0(\mathbb{Q}_p)$ isomorphic to $\mathbb{Q}_p$.
\end{proposition}
\begin{proof}
Define a map $\psi:\mathbb{Q}_p\rightarrow{N}_{r,k}$ by $\psi(a):=N_{r,k}(a)$ for all $a\in\mathbb{Q}_p$.
Let $a,b\in\mathbb{Q}_p$, then $\psi(a)\psi(b)=N_{r,k}(a)N_{r,k}(b)$ is a unitriangular matrix with the block $M_{1,r}=(m_{ij})$ that contains the elements $m_{k,k}=a+b$ and $m_{k+r,k+1}=-(a+b)$, thus $\psi(a)\psi(b)=\psi(a+b)$ and $\ker\psi=0$. 
\end{proof}

\subsubsection{The Reductive subgroup}
Let $h\in{G}_n^0(\mathbb{Q}_p)$ be an $\mathcal{L}$-automorphism defined on $\sfrac{\mathcal{L}}{\gamma_2}$ by $h(e_i)=\lambda_i{e}_i$, where $\lambda_i\in\mathbb{Q}_p$, for every $1\leq{i}\leq{n-1}$.
\begin{proposition}
For every $1\leq{i}<j\leq{n}$, $h$ must satisfy $h(e_{ij})=\lambda_i\lambda_{i+1}\cdots\lambda_{j-1}\lambda_j{e}_{ij}$.
\end{proposition}
\begin{proof}
$h(e_{ij})=h([e_{i},[e_{i+1},[\cdots[e_{j-1},e_j]]]])=[h(e_i),[h(e_{i+1}),[\cdots,[h(e_{j-1}),h(e_j)]]]]=[\lambda_i{e}_i,[\lambda_{i+1}e_{i+1},[\cdots,[\lambda_{j-1}e_{j-1},\lambda_j{e}_j]]]]=\lambda_i\lambda_{i+1}\cdots\lambda_{j-1}\lambda_j{e}_{ij}$.
\end{proof}
We conclude that $h$ is the $m\times{m}$ diagonal matrix 
\[h=\mathrm{diag}(\lambda_1,\lambda_2,\dots,\lambda_{n-1},\lambda_1\lambda_2,\dots,\lambda_2\lambda_3,\dots,\lambda_{n-2}\lambda_{n-1},\lambda_1\lambda_2\lambda_3,\dots,\lambda_1\lambda_2\cdots\lambda_{n-2}\lambda_{n-1}).\]
Denote by $H(\mathbb{Q}_p):=\{h\in{G}_n^0(\mathbb{Q}_p)\}$ the set of all diagonal $\mathcal{L}$-automorphisms.
\begin{proposition}
The set $H(\mathbb{Q}_p)$ is a subgroup of $G_n^0(\mathbb{Q}_p)$.
\end{proposition}
\begin{proof}
Let $A=\mathrm{diag}(\lambda_1,\lambda_2,\dots,\lambda_1\lambda_2,\dots)$ and $B=\mathrm{diag}(\delta_1,\delta_2,\dots,\delta_1\delta_2,\dots)$, then $AB=\mathrm{diag}(\lambda_1\delta_1,\lambda_2\delta_2,\dots\lambda_1\lambda_2\delta_1\delta_2,\dots)=\mathrm{diag}(\lambda_1\delta_1,\lambda_2\delta_2,\dots(\lambda_1\delta_1)(\lambda_2\delta_2),\dots).$
\end{proof}

\section{Computation of the integral}
We want to integrate the function $\psi:=|\det{M}|_p^s=|\lambda_1^4\lambda_2^6\lambda_3^6\lambda_4^4|_p^{s}$ over the monoid $G^+(\mathbb{Q}_p)$.
We denote by $v_1=v_p(\lambda_1),v_2=v_p(\lambda_2),v_3=v_p(\lambda_3),v_4=v_p(\lambda_4)$ the $p$-adic valuations of the elements on the main diagonal of $M_{11}$ the top-left block of the product matrix $M$.

We denote by $t=p^{-s}$, where $s$ is the complex parameter of our zeta-function.

Thus, $\psi=|\lambda_1^4\lambda_2^6\lambda_3^6\lambda_4^4|_p^s=(|\lambda_1|_p^4|\lambda_2|_p^6|\lambda_3|_p^6|\lambda_4|_p^4)^s=(p^{-v_1})^{4s}(p^{-v_2})^{6s}(p^{-v_3})^{6s}(p^{-v_4})^{4s}=(p^{-s})^{4v_1}(p^{-s})^{6v_2}(p^{-s})^{6v_3}(p^{-s})^{4v_4}=t^{4v_1}t^{6v_2}t^{6v_3}t^{4v_4}$.

Now we need to define our domain of integration, and calculate its Haar measure. Since we are computing a $p$-adic integral, we multiply the function to integrate by the domain measure, the result being our desired integral.\\ 
In the all following, we may write \textbf{constraint $m_{i,j}$} to denote a constraint implied by the matrix element $m_{i,j}$.
\subsection{Computation of constraints on $c_1,c_2,c_3,c_4$}
We compute the constraints on $c_1,c_2,c_3,c_4$ by considering the following product matrix elements:
\begin{flalign*}
m_{1,10}&=c_1\lambda_2\lambda_3\lambda_1\lambda_1\in\mathbb{Z}_p\iff{v(c_1\lambda_2\lambda_3\lambda_1\lambda_1)=v(c_1)+v(\lambda_1)+v(\lambda_2)+v(\lambda_3)+v(\lambda_4)\geq{0}}&\\
m_{2,10}&=c_2\lambda_2\lambda_3\lambda_1\lambda_1\in\mathbb{Z}_p\iff{v(c_2\lambda_2\lambda_3\lambda_1\lambda_1)=v(c_2)+v(\lambda_1)+v(\lambda_2)+v(\lambda_3)+v(\lambda_4)\geq{0}}&\\
m_{3,10}&=c_3\lambda_2\lambda_3\lambda_1\lambda_1\in\mathbb{Z}_p\iff{v(c_3\lambda_2\lambda_3\lambda_1\lambda_1)=v(c_3)+v(\lambda_1)+v(\lambda_2)+v(\lambda_3)+v(\lambda_4)\geq{0}}&\\
m_{4,10}&=c_4\lambda_2\lambda_3\lambda_1\lambda_1\in\mathbb{Z}_p\iff{v(c_4\lambda_2\lambda_3\lambda_1\lambda_1)=v(c_4)+v(\lambda_1)+v(\lambda_2)+v(\lambda_3)+v(\lambda_4)\geq{0}}&
\end{flalign*}
Hence we obtain the following constraints:
\[
v(c_1),v(c_2),v(c_3),v(c_4)\geq-(v_1+v_2+v_3+v_4)\Rightarrow\mu(c_1),\mu(c_2),\mu(c_3),\mu(c_4)=p^{v_1+v_2+v_3+v_4}.\]
Hence we multiply our integral $4$ times by the same measure $p^{v_1+v_2+v_3+v_4}$, to obtain:
\[
\int_{\underline\lambda,\underline{a},\underline{b}}\psi{p}^{4v_1+4v_2+4v_3+4v_4}d\mu(\underline\lambda)d\mu(\underline{a})d\mu(\underline{b}).
\]

m_{1,8}&=b_{11}\lambda_1\lambda_2\lambda_3\in\mathbb{Z}_p\Rightarrow{v(b_{11})\geq-(v_1+v_2+v_3)}&\\

$\alpha=v_1+v_2+v_3$.

\begin{enumerate}

But all this shows that $\left(a+p^{-\beta}\mathbb{Z}_p\right)\subseteq{p^{-\alpha}\mathbb{Z}_p}$, which means that \\
$\left(a+p^{-\beta}\mathbb{Z}_p\right)\cap{p^{-\alpha}\mathbb{Z}_p}=\left(a+p^{-\beta}\mathbb{Z}_p\right)$.

Hence $\mu\left(\left(a+p^{-\beta}\mathbb{Z}_p\right)\cap{p^{-\alpha}\mathbb{Z}_p}\right)=\mu\left(\left(a+p^{-\beta}\mathbb{Z}_p\right)\right)=\mu(p^{-\beta}\mathbb{Z}_p)=p^{\beta}=p^{v_2+v_3+v_4}$.\\

But all this shows that $x$ cannot be both in $p^{-\alpha}\mathbb{Z}_p$ and in $\left(a+p^{-\beta}\mathbb{Z}_p\right)$, hence $p^{-\alpha}\mathbb{Z}_p\cap\left(a+p^{-\beta}\mathbb{Z}_p\right)=\phi$.
\end{enumerate}
The above is symmetrical to $\beta\geq\alpha\iff-\alpha\geq-\beta$, because if $\beta\geq-v_p(a)\iff{v_p(a)}\geq-\beta$, then $v_p(x)\geq\min\{v_p(a),-\beta\}=-\beta$, but $x\in{p}^{-\alpha}\mathbb{Z}_p\iff{v(x)}\geq-\alpha\geq-\beta\implies{p}^{}$.\\

\[\mu(b_{11})=

We use the following summation rules:
\begin{align*}
&\sum_{a\geq{0}}x^a=\frac{1}{1-x}.\\
