\section{Rigidity of the Algebra}
Let $L$ be some $K$-algebra, and let $I<L$ be an ideal of $L$, We denote by $C_{\sfrac{L}{I}}(x):=\{y\in{L} : [x,y]\in{I}\}$, if $I=0$ then we call $C_L(x):\{y\in{L} : [y,x]=0\}$ the centralizer of $x$ in $L$, and if $X\subseteq{L}$ then $C_{\sfrac{L}{I}}(X) : \{y\in{L} : [y,X]\subseteq{I}\}$.\\
Denote by $L=\mathcal{L}_{n,p}$ our usual algebra over $K$.\\
\subsection{The $\mathcal{X}$ criterion}
Let $\mathcal{X}:=\{x\in{L}\setminus\gamma_2L : C_{\sfrac{L}{\gamma_3}}(x)=Kx+\gamma_2L\}$, we need to verify that $\mathcal{X}$ generates the entire algebra $L$.\\
Let $x\in{L}\setminus\gamma_2{L}$, that is\\  \[x=a_1e_{12}+a_2e_{23}+\cdots+a_{n-1}e_{n-1,n}+\sum_{r=2}^{n-1}\sum_{i=1}^{n-r}a_{i,i+r}e_{i,i+r},\] 
where $a_1,a_2,\dots,a_{n-1}$ are not all zero.\\
Let $y\in{C}_{\sfrac{L}{\gamma_3}}(x)\cap\left(L\setminus\gamma_2L\right)$, that is 
\[y=b_1e_{12}+b_2e_{23}+\cdots+b_{n-1}e_{n-1,n}+\sum_{r=2}^{n-1}\sum_{i=1}^{n-r}b_{i,i+r}e_{i,i+r},\] 
where $b_1,b_2,\dots,b_{n-1}$ are not all zero, such that $[x,y]\in\gamma_3L$.\\
But $[b_{i,i+r}e_{i,i+r},x]\in\gamma_3L$, for every $2\leq{r}\leq{n-1}$ and $1\leq{i}\leq{n-r}$. Thus
\[
[x,y]=\sum_{i=1}^{n-2}\left(a_ib_{i+1}-a_{i+1}b_i\right)e_{i,i+2}+\gamma_3L
,\]
hence $[x,y]\in\gamma_3L\iff{a}_ib_{i+1}-a_{i+1}b_i=0\iff\det\begin{pmatrix}
a_i&a_{i+1}\\
b_i&b_{i+1}
\end{pmatrix}=0$, for every $1\leq{i}\leq{n-2}$.\\
Assume $a_i\neq{0}$ for every $1\leq{i}\leq{n-1}$.\\
consider the case where $b_j=0$ for some $1\leq{j}\leq{n-1}$, then\\ $a_{j-1}b_j-a_jb_{j-1}=0\iff{b}_{j-1}=0$, and $a_{j}b_{j+1}-a_{j+1}b_j=0\iff{b}_{j+1}=0$, because $a_{j-1},a_j,a_{j+1}\neq{0}$, which implies that either $b_i\neq{0}$ for every $1\leq{i}\leq{n-1}$ or $y\in\gamma_3L$. Hence we also assume that $b_1,b_2,\dots,b_{n-1}$ are all nonzero elements in $K$. But then\\
\[a_{i}b_{i+1}-a_{i+1}b_i=0\iff\det\begin{pmatrix}
a_i&a_{i+1}\\
b_i&b_{i+1}
\end{pmatrix}=0\iff\begin{pmatrix}
    b_i&b_{i+1}
\end{pmatrix}=c_i\begin{pmatrix}
    a_i&a_{i+1}
\end{pmatrix},\]
for some nonzero element $c_i\in{K}$. Thus\\ 
\begin{align*}
&a_1b_2-a_2b_1=0&\iff&\begin{pmatrix}
    b_1&b_2
\end{pmatrix}=c_1\begin{pmatrix}
    a_1&a_2
\end{pmatrix}.\\
&a_2b_3-a_3b_2=0&\iff&\begin{pmatrix}
    b_2&b_3
\end{pmatrix}=c_2\begin{pmatrix}
    a_2&a_3
\end{pmatrix}.\\
&\vdots&\vdots\\
&a_{n-2}b_{n-1}-a_{n-1}b_{n-2}=0&\iff&\begin{pmatrix}
    b_{n-2}&b_{n-1}
\end{pmatrix}=c_{n-2}\begin{pmatrix}
    a_{n-2}&a_{n-1}
\end{pmatrix},
\end{align*}
for nonzero elements $c_1,c_2,\dots,c_{n-2}\in{K}$.\\
But then $c_{i+1}=\frac{b_{i+2}}{a_{i+2}}=\frac{b_{i+1}}{a_{i+1}}$ and $c_{i}=\frac{b_{i+1}}{a_{i+1}}=\frac{b_i}{a_i}$, for every $1\leq{i}\leq{n-2}$, which implies that $c_1=c_2=\cdots=c_{n-2}=c$ for some nonzero element $c\in{K}$, hence \[y=\sum_{i=1}^{n-1}b_ie_{i,i+1}+\gamma_2L=\sum_{i=1}^{n-1}ca_ie_{i,i+1}+\gamma_2L=c\sum_{i=1}^{n-1}a_ie_{i,i+1}+\gamma_2L=cx+\gamma_2L,\]
which means that $y\in{K}x+\gamma_2L$ for every $x\in{L}$, because the above is true for every $c\in{K}$.
But we assumed that $a_i\neq{0}$ for every $1\leq{i}\leq{n-1}$, hence every $x=a_1e_{12}+\cdots+a_{n-1}e_{n-1,n}$, where all the $a$ coefficients are nonzero elements in $K$, is in $\mathcal{X}$.\\ 
Hence, we choose $n-1$ linearly-independent vectors of the form \[x_j=\sum_{i=1}^{n-1}a_{ji}e_{i,i+1}\in\mathcal{X},\]
where $a_{ji}\neq{0}$ for every $1\leq{i,j}\leq{n-1}$, and thus
\[e_{i,i+1}=\sum_{j=1}^{n-1}a_{ij}x_j,\]
for all $1\leq{i}\leq{n-1}$, which shows that $\langle
{x}_1,x_2,\dots,x_{n-1}\rangle=\langle{e}_{12},e_{23},\dots,e_{n-1,n}\rangle$.\\
Assume $a_i=0$ for some $2\leq{i}\leq{n-2}$, and assume that $a_j\neq{0}$ for every $j\neq{i}$, then $a_ib_{i+1}-a_{i+1}b_i=-a_{i+1}b_i=0$ and $a_{i-1}b_i-a_ib_{i-1}=a_{i-1}b_i=0$ if and only if $b_i=0$. But then $\frac{b_1}{a_1}=\frac{b_2}{a_2}=\cdots=\frac{b_{i-1}}{a_{i-1}}=c_1$ and $\frac{b_{i+1}}{a_{i+1}}=\frac{b_{i+2}}{a_{i+2}}=\cdots=\frac{b_{n-1}}{a_{n-1}}=c_2$, for $c_1,c_2\in{K}$, hence $y=c_1(a_1e_{1,2}+a_2e_{2,3}+\cdots+a_{n-1}e_{i-1,i})+c_2(a_{i+1}e_{i+1,i+2}+a_{i+2}e_{i+2,i+3}+\cdots+a_{n-1}e_{n-1,n})+\gamma_2L\notin{K}x+\gamma_2L\implies{x}\notin\mathcal{X}$, which is an example for an element whose centralizers are not necessarily in $Kx+\gamma_2L$.
\subsection{The $\mathcal{Y}$ criterion}
Let $x=a_ie_{i,i+1}$ for some $1\leq{i}\leq{n-1}$, and let a centralizer element be $y=b_1e_{12}+\cdots+b_{1,n-1}e_{1,n-1}\in{C}_L(x)$, for some $b_1,\dots,b_{1,n-1}\in{K}$ not all zero.\\
But \[[x,y]=0\iff\sum_{k=1}^{r-1}a_{j,j+k}b_{j+k,j+r}-a_{j+r,j+k}b_{j,j+k}=0,\]
for all $2\leq{r}\leq{n-1}$ and $1\leq{j}\leq{n}-r$.\\
But every expression of the form $a_{jk}b_{kl}$ vanishes, because every $a_{jk}=0$ where $(j,k)\neq(i,i+1)$, which implies that $a_{i,i+1}b_{i+1,r}=0\iff{b_{i+1,r}=0}$ for all $i+2\leq{r}\leq{n}$, and that $a_{i,i+1}b_{k,i}=0\iff{b_{k,i}=0}$ for all $1\leq{k}\leq{i-1}$.\\
Hence, a generic $y\in{C}_L(x)$ has the form
\[
y=\sum_{r=1}^{n-1}\left[\sum_{j=1}^{i-r-1}b_{j,j+r}e_{j,j+r}+b_{i,i+r}e_{i,i+r}+\sum_{j=i+2}^{n-r}b_{j,j+r}e_{j,j+r}\right],
\]
such that the $b$ coefficients are not all zero.\\
Let $z=d_1e_{12}+\cdots+d_{2,n}e_{2,n}\in{C}_L(y)$, and denote\\
\begin{flalign*}
\varphi_{13}&=(b_1d_2-b_2d_1)e_{13}.\\
\varphi_{24}&=(b_2d_3-b_3d_2)e_{24}.\\
\vdots\\
\varphi_{n-2,n}&=(b_{n-2}d_{n-1}-b_{n-1}d_{n-2})e_{n-2,n}.\\
\varphi_{14}&=(b_1d_{24}-b_{24}d_1+b_{13}d_3-b_3d_{13})e_{14}.\\
\varphi_{25}&=(b_2d_{35}-b_{35}d_2+b_{24}d_4-b_4d_{24})e_{25}.\\
\vdots\\
\varphi_{n-3,n}&=(b_{n-3}d_{n-2,n}-b_{n-2,n}d_{n-3}+b_{n-3,n-1}d_{n-1}-b_{n-1}d_{n-3,n-1})e_{n-3,n}.\\
\vdots\\
\varphi_{1n}&=(b_1d_{2,n}-b_{2,n}d_1+b_{13}d_{3,n}-b_{3,n}d_{13}+\cdots+b_{1,n-1}d_{n-1}-b_{n-1}d_{1,n-1})e_{1n}.\\
\end{flalign*}
Hence
\[[z,y]=\sum_{r=2}^{n-1}\sum_{j=1}^{n-r}\varphi_{j,j+r}=0,\]
implies that $\varphi_{j,j+r}=0$
for all $2\leq{r}\leq{n-1}$ and $2\leq{j}\leq{n-r}$.\\
Let $d_{lh}\in{K}$ for some $2\leq{l}\leq{h-1}$, where $(l,h)\neq(i,i+1)$, then for some $1\leq{k}\leq{h-2}$, we have
\[\varphi_{kh}=\sum_{r=1}^{h-k-1}\left(b_{k,k+r}d_{k+r,h}-b_{k+r,h}d_{k,k+r}\right)e_{kh},\]
where $b_{kl}d_{lh}$ is one of the expressions in the sum. But all other expressions $b_{kq}d_{qh}$, where $q\neq{l}$, vanish because $b_{kq}=0$. But this implies that\\ $\varphi_{kh}=0\iff{b}_{kl}d_{lh}=0\iff{d}_{lh}=0$, which means that if $y_{kl}=b_{kl}e_{kl}$ and $z_{kl}=d_1e_{i,i+1}+\cdots+d_{1n}e_{1n}\in{C}_L(y_{kl})$, then $d_{lh}=0$.\\
We can also show that $d_{lh}e_{lh}$ must vanish in any $z_{hk}\in{C}_L(y_{hk})$ by considering 
\[\varphi_{lk}=\sum_{r=1}^{k-l-1}\left(b_{l,l+r}d_{l+r,k}-b_{l+r,k}d_{l,l+r}\right)e_{lk},\]
for some $l+2\leq{k}\leq{n}$, and looking at $b_{hk}d_{lh}$.\\
For $d_i$, we observe that since $b_{ki}=b_{i+1,l}=0$ for every $1\leq{k}<i$ and every $i+1<l\leq{n}$, then $b_{ki}d_i=0$ or $b_{i+1,l}d_i=0$ does not imply that $d_i=0$, so $d_i\in{K}$ can be any nonzero element such that
\begin{flalign*}
[b_{ki}e_{ki},d_ie_{i,i+1}]&=0.\\ [b_{i+1,l}e_{i+1,l},d_ie_{i,i+1}]&=0.\\
\end{flalign*}
But all this implies that if $B=\{b_{kl} : 1\leq{k}<l\leq{n},\,l\neq{i},\,k\neq{i+1}\}$, then any $z\in{C}_L(B)$ must have $d_{lh}=0$ for all $1\leq{l}<h\leq{n}$, unless $(l,h)=(i,i+1)$, but then $z=d_ie_{i,i+1}=ca_ie_{i,i+1}=cx$ for some $c=\frac{d_i}{a_i}\in{K}$, and all the computation above is up to the center $Z(L)$, hence $z\in{K}x+Z(L
)$. But this is true for all $1\leq{i}\leq{n-1}$, which shows that if $E=\{e_{i,i+1},e_{i+1,i+2},\dots,e_{n-1,n}\}$ is the generating set of $L$, then $E\subseteq\mathcal{Y}$, but then we conclude from both criteria $\mathcal{X}$ and $\mathcal{Y}$ that $L$ is rigid.

\subsubsection{The Reductive subgroup}
Let $\varphi\in{G}_n^0(K)$ be an $\mathcal{L}$-automorphism formed as a matrix $a=(a_{ij})$, and let $h$ be the matrix with only the main diagonal of $a$, that is \[h=\mathrm{diag}(a_{11},a_{22},\dots,a_{n-1,n-1}),a_{11}a_{22},\dots,a_{n-3}a_{n-2},a_{n-2}a_{n-1},\dots,a_{11}a_{22}\cdots{a}_{n-1}),\]
and it is readily seen that $h$ itself is an $\mathcal{L}$-automorphism.\\
Denote by $H(K):=\{h\in{G}_n^0(K)\}$ the set of all diagonal $\mathcal{L}$-automorphisms.
\begin{proposition}
The set $H(K)$ is a subgroup of $G_n^0(K)$.
\end{proposition}
\begin{proof}
Let $A=\mathrm{diag}(\lambda_1,\lambda_2,\dots,\lambda_1\lambda_2,\dots)$ and $B=\mathrm{diag}(\delta_1,\delta_2,\dots,\delta_1\delta_2,\dots)$, for some $\lambda_1,\lambda_2,\dots,\lambda_{n-1},\delta_1,\delta_2,\dots,\delta_{n-1}\inK$, then $AB=\mathrm{diag}(\lambda_1\delta_1,\lambda_2\delta_2,\dots\lambda_1\lambda_2\delta_1\delta_2,\dots)=\mathrm{diag}(\lambda_1\delta_1,\lambda_2\delta_2,\dots(\lambda_1\delta_1)(\lambda_2\delta_2),\dots).$
\end{proof}
A simple check would show that $H(K)$ is not normal in $G_n^0(K)$.

\subsection{Basic Decomposition of $\mathcal{L}$-automorphisms}
\subsubsection{The Unipotent Radical}
Let $a$ and $h$ be an arbitrary automorphism and a diagonal automorphism, respectively, and let $u=h^{-1}a$, then $a=uh$.
\begin{proposition}
The matrix $u=(u_{ij})$ is unitriangular.
\end{proposition}
\begin{proof}
$h=\mathrm{diag}(a_{11},a_{22},\dots)$, which implies that $h^{-1}=\mathrm{diag}(a_{11}^{-1},a_{22}^{-1},\dots)$, thus $u=h^{-1}a$ has $u_{ii}=a_{ii}^{-1}a_{ii}=1$, for all $1\leq{i}\leq{d}$.
\end{proof}
From the construction, it is immediate to assert that every $\varphi\in{G}_n^0(K)$ has the unique factorization $\varphi=uh$, an we denote the set $N_n(K):=\{u\in{G}_n^0(K)\}$, the set of all unitriangular $\mathcal{L}$-automorphisms.
\begin{proposition}
The set $N_n(K)$ is a normal subgroup of $G_n^0(K)$.
\end{proposition}
\begin{proof}
The product of two unitriangular matrices is itself a triangular matrix. Let $w\in{G}_n^0(K)$ a unitriangular $\mathcal{L}$-automorphism. For any $a\in{G}_n^0$, we have the unique factorization $a=uh$, for some $u\in{N}_n(K)$ and $h\in{H}_n(K)$, thus $awa^{-1}=(uh)w(uh)^{-1}=u(hwh^{-1})u^{-1}$, but then $b=hwh^{-1}=(b_{ij})$ has $b_{ii}=h_{ii}\cdot{1}\cdot{h}_{ii}^{-1}=1$, for all $1\leq{i}\leq{d}$, hence $b$ and therefore $ubu^{-1}$ also belong to $N_n(K)$.
\end{proof}
\begin{corollary}
The group of diagonal $\mathcal{L}$-automorphisms has a basic decomposition $G_n^0(K)=N_n(K)\rtimes{H}_n(K)$.
\end{corollary}
This is an immediate result, because $N_n(K)\cap{H}_n(K)=\{I_d\}$.

\subsubsubsection{The Structure of $\varphi_{n-2}$}
\begin{proposition}
Let $r=n-2$, and denote the $(1,n-2)$ block of $\varphi_{n-2}$ by $M_{1,n-2}=(m_{ij})$. Then the only nonzero elements in $M_{1,n-2}$ are $m_{11}$, $m_{12}$, $m_{22}$, $m_{n-2,1}$, $m_{n-1,1}$ and $m_{n-1,2}$, and in addition $m_{n-1,2}=-m_{11}$.
\end{proposition}
$M_{1,n-2}=\begin{bmatrix}
m_{11}&m_{12}\\
&m_{22}\\
\vdots&\vdots\\
m_{n-2,1}&\\
m_{n-1,1}&-m_{11}\\

\begin{proof}
We have the following constraints\\
$\varphi_{n-2}([e_1,e_{1+k}])=[\varphi_{n-2}(e_1),\varphi_{n-2}(e_{1+k})]=m_{1+k,2}[e_1,e_{2,n}]$, but $[e_1,e_{1+k}]=0$ for all $2\leq{k}\leq{n-3}$, which implies that $m_{1+k,2}=0$.\\

From this it is clear that the only elements that are not affected by these constraints are the ones listed above.\\
Also, since $[\varphi_{n-2}(e_1),\varphi_{n-2}(e_{n-1})]=[e_1,m_{n-1,2}e_{2,n}]+[m_{11}e_{1,n-1}e_{n-1}]=\\=(m_{n-1,2}+m_{11})e_{1,n}=0$, we conclude that $m_{n-1,2}=-m_{11}$.
\end{proof}

\section{Full Decomposition of the Zeta-Function}
We aim at computing $\zeta_{L_{n,p}}^{\wedge}(s)=\displaystyle\int_{G_{n}^{+}(\mathbb{Q}_p)}|\det{\varphi}|_p^sd\mu_{G_{n}(\mathbb{Z}_p)\varphi}$, which is the product of the function $|det\varphi|_p^s$ by the measure of domain of integration.
\subsection{Computation of the Function Under the Integral}
\begin{proposition}
Let $n\geq{4}$, and $\varphi\in{G}_n^0(\mathbb{Z}_p)$ for some prime $p$. As seen above, $\varphi$ has the decomposition $\varphi=uh$, where $u\in{N}_n(\mathbb{Z}_p)$ and $h\in{H}_n(\mathbb{Z}_p)$. Then $|\det\varphi|_p^s=\prod_{i=1}^{n-1}|\lambda_i|_p^{i(n-i)s}$, where $\lambda_i=h_{ii}$ for $1\leq{i}\leq{n}-1$.
\begin{proof}
Since $\det{u}=1$, it follows that $\det\varphi=\det{h}$, which is the product of all the occurrences of $\lambda_1,\lambda_2,\dots,\lambda_{n-1}$ in $h$, which is $\#\lambda_i(n)=i(n-i)$. This is a simple combinatorial formula that can be proved for $n=5$ by counting the number of $\lambda_1$, $\lambda_2$, $\lambda_3$ and $\lambda_4$, which are $4$, $6$, $6$ and $4$, respectively. For $n+1$, we observe that for each length $l$ we add a new product $\lambda_{n-l+1}\cdots\lambda_{n-1}\lambda_n$, thus we add $i$ times each element $\lambda_i$, which aligns with the formula, because $\#\lambda_i(n+1)=i(n+1-i)=i(n-i)+i=\#\lambda_i(n)+i$.
\end{proof}
\end{proposition}
\subsection{Full Decomposition of the Domain of Integration}
By applying Fubini's theorem for semidirect products of topological groups \cite[Proposition 28]{Nachbin}, we have that \[\zeta_{L_{n,p}}^{\wedge}(s)=\displaystyle\int_{G_{n}^{+}(\mathbb{Q}_p)}|\det{\varphi}|_p^sd\mu_{G_{n}(\mathbb{Z}_p)\varphi}=\displaystyle\int_{H^+(\mathbb{Q}_p)}\left(\displaystyle\int_{N_{h}^+}|\det{uh}|_p^sd\mu_{N(\mathbb{Q}_p)}\right)d\mu_{H(\mathbb{Q}_p)},\]
where \[H^+(\mathbb{Q}_p):=\{\mathrm{diag}(\lambda_{1},\dots,\lambda_{n-1},\lambda_{1}\lambda_{2},\dots,\lambda_{1}\lambda_{2}\cdots\lambda_{n-2}\lambda_{n-1}) : \lambda_{i}\in\mathbb{Z}_{p}\setminus\{0\}\},\]
that is, $H^+(\mathbb{Q}_p)$ consists of all $h\in{H(\mathbb{Q}_{p})}$ that appear in the decomposition $\varphi=uh$ for some $\varphi\in{G_{n}^{+}(\mathbb{Q}_{p})}$, and, for a given $h\in{H^{+}(\mathbb{Q}_{p})}$, we set $N_{h}^{+}:=\{u\in{N(\mathbb{Q}_{p}) : uh\in{G_{n}^{+}(\mathbb{Q}_{p})}}\}$. The integrand is constant on $N_{h}^{+}$, so computing the inner integral amounts to finding the measure of $N_{h}^+$, which is complicated, but using the decomposition from Corollary \ref{cor:Nr.decomposition}, we can simplify $N_{h}^{+}$ at the price of replacing a single integral by multiple integrals.\\
Let $\mathcal{N}_{1},\mathcal{N}_{2},\dots,\mathcal{N}_{m}$, where $m=\binom{n}{2}$, be an enumeration of the subgroups
\[N_{2,n-2},N_{2,n-3},\dots,N_{2,0},N_{3,n-3},\dots,N_{3,0},\dots\]\[\dots,N_{n-2,2},N_{n-2,1},N_{n-2,0},N'_{n-2},N''_{n-2},N_{n-1}.\]
Every $\varphi\in{G_{n}^{0}(\mathbb{Q}_{p})}$ can be written uniquely as $\varphi=u_{m}\cdots{u_{1}h}$, where $u_{i}\in{\mathcal{N}_{i}}$. Thus, by Fubini \[
\zeta_{L_{n,p}}^{\wedge}(s)=\displaystyle\int_{H^{+}}\displaystyle\int_{\mathcal{N}_{1}^{+}(h)}\displaystyle\int_{\mathcal{N}_{2}^{+}(h,u_{1})}\cdots\displaystyle\int_{\mathcal{N}_{m}^{+}(h,u_{1},\dots,u_{m-1})}|\det{h}|_{p}^{s}d\mu_{H}d\mu_{\mathcal{N}_{1}}\cdots{d\mu_{\mathcal{N}_{m}}},\]
where each $\mu_{\mathcal{N}_{i}}$ is Haar measure on $\mathcal{N}_{i}(\mathbb{Q}_{p})$ normalized so that\\ $\mu_{\mathcal{N}_{i}}(\mathcal{N}_{i}(\mathbb{Z}_{p}))=1$, and \[\mathcal{N}_{i}^{+}(h,u_{1},\dots,u_{i-1}):=\{u_{i}\in\mathcal{N}_{i} : \exists{u_{i+1},\dots,u_{m}}\,\mathrm{such}\,\mathrm{that}\,u_{m}\cdots{u_{1}}h\in{G_{n}^{+}(\mathbb{Q}_{p})}\}.\]
\section{Computation of the integral}
We want to integrate the function $\psi:=|\det{M}|_p^s=|\lambda_1^4\lambda_2^6\lambda_3^6\lambda_4^4|_p^{s}$ over the monoid $G^+(K)$.
We denote by $v_1=v_p(\lambda_1),v_2=v_p(\lambda_2),v_3=v_p(\lambda_3),v_4=v_p(\lambda_4)$ the $p$-adic valuations of the elements on the main diagonal of $M_{11}$ the top-left block of the product matrix $M$.

We denote by $t=p^{-s}$, where $s$ is the complex parameter of our zeta-function.

Thus, $\psi=|\lambda_1^4\lambda_2^6\lambda_3^6\lambda_4^4|_p^s=(|\lambda_1|_p^4|\lambda_2|_p^6|\lambda_3|_p^6|\lambda_4|_p^4)^s=(p^{-v_1})^{4s}(p^{-v_2})^{6s}(p^{-v_3})^{6s}(p^{-v_4})^{4s}=(p^{-s})^{4v_1}(p^{-s})^{6v_2}(p^{-s})^{6v_3}(p^{-s})^{4v_4}=t^{4v_1}t^{6v_2}t^{6v_3}t^{4v_4}$.

Now we need to define our domain of integration, and calculate its Haar measure. Since we are computing a $p$-adic integral, we multiply the function to integrate by the domain measure, the result being our desired integral.\\
