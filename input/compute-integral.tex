\section{Computation of the integral}
We want to integrate the function $\psi:=|\det{M}|_p^s=|\lambda_1^4\lambda_2^6\lambda_3^6\lambda_4^4|_p^{s}$ over the monoid $G^+(\mathbb{Q}_p)$.
We denote by $v_1=v_p(\lambda_1),v_2=v_p(\lambda_2),v_3=v_p(\lambda_3),v_4=v_p(\lambda_4)$ the $p$-adic valuations of the elements on the main diagonal of $M_{11}$ the top-left block of the product matrix $M$.

We denote by $t=p^{-s}$, where $s$ is the complex parameter of our zeta-function.

Thus, $\psi=|\lambda_1^4\lambda_2^6\lambda_3^6\lambda_4^4|_p^s=(|\lambda_1|_p^4|\lambda_2|_p^6|\lambda_3|_p^6|\lambda_4|_p^4)^s=(p^{-v_1})^{4s}(p^{-v_2})^{6s}(p^{-v_3})^{6s}(p^{-v_4})^{4s}=(p^{-s})^{4v_1}(p^{-s})^{6v_2}(p^{-s})^{6v_3}(p^{-s})^{4v_4}=t^{4v_1}t^{6v_2}t^{6v_3}t^{4v_4}$.

Now we need to define our domain of integration, and calculate its Haar measure. Since we are computing a $p$-adic integral, we multiply the function to integrate by the domain measure, the result being our desired integral.\\ 
In the all following, we may write \textbf{constraint $m_{i,j}$} to denote a constraint implied by the matrix element $m_{i,j}$.
\subsection{Computation of constraints on $c_1,c_2,c_3,c_4$}
We compute the constraints on $c_1,c_2,c_3,c_4$ by considering the following product matrix elements:
\begin{flalign*}
m_{1,10}&=c_1\lambda_2\lambda_3\lambda_1\lambda_1\in\mathbb{Z}_p\iff{v(c_1\lambda_2\lambda_3\lambda_1\lambda_1)=v(c_1)+v(\lambda_1)+v(\lambda_2)+v(\lambda_3)+v(\lambda_4)\geq{0}}&\\
m_{2,10}&=c_2\lambda_2\lambda_3\lambda_1\lambda_1\in\mathbb{Z}_p\iff{v(c_2\lambda_2\lambda_3\lambda_1\lambda_1)=v(c_2)+v(\lambda_1)+v(\lambda_2)+v(\lambda_3)+v(\lambda_4)\geq{0}}&\\
m_{3,10}&=c_3\lambda_2\lambda_3\lambda_1\lambda_1\in\mathbb{Z}_p\iff{v(c_3\lambda_2\lambda_3\lambda_1\lambda_1)=v(c_3)+v(\lambda_1)+v(\lambda_2)+v(\lambda_3)+v(\lambda_4)\geq{0}}&\\
m_{4,10}&=c_4\lambda_2\lambda_3\lambda_1\lambda_1\in\mathbb{Z}_p\iff{v(c_4\lambda_2\lambda_3\lambda_1\lambda_1)=v(c_4)+v(\lambda_1)+v(\lambda_2)+v(\lambda_3)+v(\lambda_4)\geq{0}}&
\end{flalign*}
Hence we obtain the following constraints:
\[
v(c_1),v(c_2),v(c_3),v(c_4)\geq-(v_1+v_2+v_3+v_4)\Rightarrow\mu(c_1),\mu(c_2),\mu(c_3),\mu(c_4)=p^{v_1+v_2+v_3+v_4}.\]
Hence we multiply our integral $4$ times by the same measure $p^{v_1+v_2+v_3+v_4}$, to obtain:
\[
\int_{\underline\lambda,\underline{a},\underline{b}}\psi{p}^{4v_1+4v_2+4v_3+4v_4}d\mu(\underline\lambda)d\mu(\underline{a})d\mu(\underline{b}).
\]
\subsection{Computation of constraints on the $b$ elements}
We consider the following product matrix elements:
\begin{flalign*}
m_{1,8}&=b_{11}\lambda_1\lambda_2\lambda_3\in\mathbb{Z}_p\Rightarrow{v(b_{11})\geq-(v_1+v_2+v_3)}&\\
m_{1,9}&=b_{12}\lambda_2\lambda_3\lambda_4\in\mathbb{Z}_p\Rightarrow{v(b_{12})\geq-(v_2+v_3+v_4)}&\\
m_{2,9}&=b_{22}\lambda_2\lambda_3\lambda_4\in\mathbb{Z}_p\Rightarrow{v(b_{22})\geq-(v_2+v_3+v_4)}&\\
m_{3,8}&=b_{31}\lambda_1\lambda_2\lambda_3\in\mathbb{Z}_p\Rightarrow{v(b_{31})\geq-(v_1+v_2+v_3)}&\\
m_{4,8}&=b_{41}\lambda_1\lambda_2\lambda_3\in\mathbb{Z}_p\Rightarrow{v(b_{41})\geq-(v_1+v_2+v_3)}&\\
m_{5,10}&=b_{22}\lambda_1\lambda_2\lambda_3\lambda_4\in\mathbb{Z}_p\Rightarrow{v(b_{22})\geq-(v_1+v_2+v_3+v_4)}&\\
m_{7,10}&=b_{31}\lambda_1\lambda_2\lambda_3\lambda_4\in\mathbb{Z}_p\Rightarrow{v(b_{31})\geq-(v_1+v_2+v_3+v_4)}&\\
\end{flalign*}
But since $v_1,v_2,v_3,v_4\geq{0}$, then $x\geq-(v_1+v_2+v_3)\implies{x}\geq-(v_1+v_2+v_3+v_4)$ and $x\geq-(v_2+v_3+v_4)\implies{x}\geq-(v_1+v_2+v_3+v_4)$, hence $m_{5,10}$ is implied by $m_{2,9}$ and $m_{7,10}$ is implied by $m_{3,8}$.
\subsubsection{Integration by $b_{12},b_{22},b_{31},b_{41}$}
By the listed constraints we obtain the measures:
\begin{flalign*}
\mu(b_{12})&=p^{v_2+v_3+v_4}.&\\
\mu(b_{22})&=p^{v_2+v_3+v_4}.&\\
\mu(b_{31})&=p^{v_1+v_2+v_3}.&\\
\mu(b_{41})&=p^{v_1+v_2+v_3}.&\\
\end{flalign*}
Hence we multiply our integral by these measures, to obtain:
\[
\int_{\underline\lambda,\underline{a},b_{11}}\psi{p}^{4v_1+4v_2+4v_3+4v_4}p^{2v_1+4v_2+4v_3+2v_4}d\mu(\underline\lambda)d\mu(\underline{a})d\mu(b_{11})=\]\[=\int_{\underline\lambda,\underline{a},b_{11}}\psi{p}^{6v_1+8v_2+8v_3+6v_4}d\mu(\underline\lambda)d\mu(\underline{a})d\mu(b_{11}).
\]
\subsubsection{Integration by $b_{11}$}
We observe that $b_{11}$ appears also in the following product matrix element:

$m_{4,9}=(a_{11}a_{22}-b_{11})\lambda_2\lambda_3\lambda_4\in\mathbb{Z}_{p}.$

Hence we have:
\begin{flalign*}
b_{11}&\in{p^{-(v_1+v_2+v_3)}}&\\
a_{11}a_{22}-b_{11}&\in{p^{-(v_2+v_3+v_4)}}\implies{b_{11}\in{a_{11}a_{22}+p^{-(v_2+v_3+v_4)}}}&
\end{flalign*}
Which means that $b_{11}\in{p^{-(v_1+v_2+v_3)}\mathbb{Z}_p}\cap\left(a_{11}a_{22}+p^{-(v_2+v_3+v_4)}\mathbb{Z}_p\right)$.

We compute $\mu\left(p^{-(v_1+v_2+v_3)}\mathbb{Z}_p\cap\left(a_{11}a_{22}+p^{-(v_2+v_3+v_4)}\mathbb{Z}_p\right)\right)$, 

the Haar measure of the intersection.

Denote:

$a=a_{11}a_{22}$.

$\alpha=v_1+v_2+v_3$.

$\beta=v_2+v_3+v_4$.

We have the following cases:
\begin{enumerate}
    \item $\alpha\geq\beta$ and $\alpha\geq-v_p(a)$.

Let $x\in{\left(a+p^{-\beta}\mathbb{Z}_p\right)\cap{p^{-\alpha}\mathbb{Z}_p}}$ be some element in the intersection, but if $x\in{a+p^{-\beta}\mathbb{Z}_p}$ then $x=a+b$, where $b\in{p^{-\beta}\mathbb{Z}_p}\iff{v_p(b)\geq{-\beta}}$, but then $v_p(x)\geq\min\{v_p(a),v_p(b)\}\geq\min\{v_p(a),-\beta\}$.

But $\alpha\geq-v_p(a)\iff{v_p(a)\geq-\alpha}\Rightarrow\min\{v_p(a),-\beta\}\geq\min\{-\alpha,-\beta\}$,

and $\alpha\geq\beta\iff-\beta\geq-\alpha\Rightarrow{v_p(x)\geq\min\{-\alpha,-\beta\}=-\alpha}\Rightarrow{x\in{p^{-\alpha}\mathbb{Z}_p}}$.

But all this shows that $\left(a+p^{-\beta}\mathbb{Z}_p\right)\subseteq{p^{-\alpha}\mathbb{Z}_p}$, which means that \\
$\left(a+p^{-\beta}\mathbb{Z}_p\right)\cap{p^{-\alpha}\mathbb{Z}_p}=\left(a+p^{-\beta}\mathbb{Z}_p\right)$.

Hence $\mu\left(\left(a+p^{-\beta}\mathbb{Z}_p\right)\cap{p^{-\alpha}\mathbb{Z}_p}\right)=\mu\left(\left(a+p^{-\beta}\mathbb{Z}_p\right)\right)=\mu(p^{-\beta}\mathbb{Z}_p)=p^{\beta}=p^{v_2+v_3+v_4}$.\\
\item $\alpha\geq\beta$ and $\alpha<-v_p(a)$.
Let $x=a+b$ be an element in the intersection as before, hence $v_p(x)=v_p(a+b)\geq\min\{v_p(a),v_p(b)\}\geq\min\{v_p(a),-\beta\}$ and $v_p(x)\geq-\alpha$.

But $\alpha<-v_p(a)\iff{v_p(a)<-\alpha}$ and $\alpha\geq\beta\iff-\beta\geq-\alpha$, hence $v_p(x)\geq\min\{v_p(a),-\beta\}\geq\min\{v_p(a),-\alpha\}=v_p(a)$. But $v_p(a)<-\alpha$ is a strong inequality, thus by a known fact for non-Archimedean valuations $v_p(x)=\min\{v_p(a),-\alpha\}=v_p(a)$.\\
But all this shows that $x$ cannot be both in $p^{-\alpha}\mathbb{Z}_p$ and in $\left(a+p^{-\beta}\mathbb{Z}_p\right)$, hence $p^{-\alpha}\mathbb{Z}_p\cap\left(a+p^{-\beta}\mathbb{Z}_p\right)=\phi$.
\end{enumerate}
The above is symmetrical to $\beta\geq\alpha\iff-\alpha\geq-\beta$, because if $\beta\geq-v_p(a)\iff{v_p(a)}\geq-\beta$, then $v_p(x)\geq\min\{v_p(a),-\beta\}=-\beta$, but $x\in{p}^{-\alpha}\mathbb{Z}_p\iff{v(x)}\geq-\alpha\geq-\beta\implies{p}^{}$.\\
Thus, we claim that:
\[\mu(b_{11})=
\begin{cases}
p^{\min\{\alpha,\beta\}},&\text{if}\,\,\,\,v_p(a)\geq-\max\{\alpha,\beta\}. \\
0,&\text{if}\,\,\,\,v_p(a)<-\max\{\alpha,\beta\}.
\end{cases}
\]
But $\max\{\alpha,\beta\}=\max\{v_1+v_2+v_3,v_2+v_3+v_4\}=v_2+v_3+\max\{v_1,v_4\}$, hence with multiply our integral by $\mu(b_{11})$, to obtain:
\[
\int_{\underline\lambda,\underline{a}}\psi{p}^{6v_1+8v_2+8v_3+6v_4}p^{v_2+v_3+\min\{v_1,v_4\}}d\mu(\underline\lambda)d\mu(\underline{a})=\]\[=\int_{\underline\lambda,\underline{a}}\psi{p}^{6v_1+9v_2+9v_3+6v_4+\min\{v_1,v_4\}}d\mu(\underline\lambda)d\mu(\underline{a}),
\]
with the constraint $v_p(a_{11}a_{22})=v_p(a_{11})+v_p(a_{22})\geq-(v_2+v_3+\min\{v_1,v_4\})$.
We denote: $v_{11}=v_p(a_{11})$, $v_{22}=v_p(a_{22})$ and $\delta=\min\{v_1,v_4\}$.
\subsection{Computation of constraints on the $a$ elements}
We have the following constraints:
\begin{flalign*}
m_{1,5}&=a_{11}\lambda_1\lambda_2\in\mathbb{Z}_p\implies{v_p(a_{11}\lambda_1\lambda_2)\geq{0}}\implies{v_{11}}\geq-(v_1+v_2)&\\
m_{2,5}&=a_{21}\lambda_1\lambda_2\in\mathbb{Z}_p\implies{v_p(a_{21}\lambda_1\lambda_2)\geq{0}}\implies{v_p(a_{21})}\geq-(v_1+v_2)&\\
m_{2,6}&=a_{22}\lambda_2\lambda_3\in\mathbb{Z}_p\implies{v_p(a_{22}\lambda_2\lambda_3)\geq{0}}\implies{v_{22}}\geq-(v_2+v_3)&\\
m_{3,6}&=-a_{11}\lambda_2\lambda_3\in\mathbb{Z}_p\implies{v_p(a_{11}\lambda_2\lambda_3)\geq{0}}\implies{v_{11}}\geq-(v_2+v_3)&\\
m_{3,7}&=a_{33}\lambda_3\lambda_4\in\mathbb{Z}_p\implies{v_p(a_{33}\lambda_3\lambda_4)\geq{0}}\implies{v_p(a_{33})}\geq-(v_3+v_4)&\\
m_{4,7}&=-a_{22}\lambda_3\lambda_4\in\mathbb{Z}_p\implies{v_p(a_{22}\lambda_3\lambda_4)\geq{0}}\implies{v_{22}}\geq-(v_3+v_4)&\\
m_{7,9}&=-a_{11}\lambda_2\lambda_3\lambda_4\in\mathbb{Z}_p\implies{v_p(-a_{11}\lambda_2\lambda_3\lambda_4)}\geq{0}\implies{v_p}(-a_{11})=v_{11}\geq-(v_2+v_3+v_4)&\\
m_{5,8}&=a_{22}\lambda_1\lambda_2\lambda_3\in\mathbb{Z}_p\implies{v_p(a_{22}\lambda_1\lambda_2\lambda_3)}\geq{0}\implies{v_p}(a_{22})\geq-(v_1+v_2+v_3)&\\
m_{6,10}&=-a_{21}a_{33}\lambda_1\lambda_2\lambda_3\lambda_4\in\mathbb{Z}_p\implies{v_p(a_{21}a_{33}\lambda_1\lambda_2\lambda_3\lambda_4)\geq{0}}\implies{v_p(a_{21})+v_p(a_{33})}\geq-(v_1+v_2+v_3+v_4)\\
m_{2,8}&=a_{21}a_{22}\lambda_1\lambda_2\lambda_3\in\mathbb{Z}_p\implies{v_p(a_{21}a_{22}\lambda_1\lambda_2\lambda_3)\geq{0}}\implies{v_{22}+v_p(a_{21})}\geq-(v_1+v_2+v_3)&\\
m_{3,9}&=-a_{11}a_{33}\lambda_2\lambda_3\lambda_4\in\mathbb{Z}_p\implies{v_p(a_{11}a_{33}\lambda_2\lambda_3\lambda_4)\geq{0}}\implies{v_{11}+v_p(a_{33})}\geq-(v_2+v_3+v_4)&\\
\end{flalign*}
We notice that the constraint $m_{6,10}$ is already implied by the constraints $m_{2,5}$ and $m_{3,7}$.
\subsubsection{Computation of constraints on $a_{21}$ and $a_{33}$}

\subsubsubsection{Integration by $a_{33}$}
We integrate by $a_{33}$ considering $a_{11}$ as a constant. We have the following constraints:
\begin{enumerate}
    \item $v_p(a_{33})\geq-v_3-v_4$.
\item $v_p(a_{33})\geq-v_{11}-v_2-v_3-v_4$.
\end{enumerate}
But $-v_{11}-v_2-v_3-v_4\geq{-v_3-v_4}\iff{-v_{11}-v_2}\geq{0}\iff{v_{11}\leq-v_2\leq{0}}$.

And $-v_{11}-v_2-v_3-v_4<{-v_3-v_4}\iff{-v_{11}-v_2}<{0}\iff{v_{11}>-v_2}$.

Hence, $v_p(a_{33})\geq-(v_3+v_4)-\min\{0,v_2+v_{11}\}$ and we multiply out integral by $\mu(a_{33})$, to obtain:
\[\underset{v_{11}+v_{22}\geq-(v_2+v_3+\delta)}{\int_{\underline\lambda,a_{11},a_{22},a_{21}}}\psi{p}^{6v_1+9v_2+6v_3+6v_4+\delta}p^{v_3+v_4}p^{\min\{0,v_2+v_{11}\}}d\mu(\underline\lambda)d\mu(a_{11})\mu(a_{22})\mu(a_{21})=\]\[=\underset{v_{11}+v_{22}\geq-(v_2+v_3+\delta)}{\int_{\underline\lambda,a_{11},a_{22},a_{21}}}\psi{p}^{6v_1+9v_2+10v_3+7v_4+\delta}p^{\min\{0,v_2+v_{11}\}}d\mu(\underline\lambda)d\mu(a_{11})\mu(a_{22})\mu(a_{21}).
\]
We denote: $\gamma=\min\{0,v_2+v_{11}\}$.
\subsubsubsection{Integration by $a_{21}$}
We integrate by $a_{21}$ considering $a_{22}$ as a constant. We have the following constraints:
\begin{enumerate}
\item $v_p(a_{21})\geq-v_1-v_2$.
    \item $v_p(a_{21})\geq-v_{22}-v_1-v_2-v_3$.
    \end{enumerate}
But $-v_{22}-v_1-v_2-v_3\geq{-v_1-v_2}\iff{-v_{22}-v_3}\geq{0}\iff{v_{22}\leq-v_3\leq{0}}$.

And $-v_{22}-v_1-v_2-v_3<{-v_1-v_2}\iff{-v_{22}-v_3}<{0}\iff{v_{22}>-v_3}$.

And hence, $v_p(a_{21})\geq-(v_1+v_2)-\min\{0,v_3+v_{22}\}$ and we multiply our integral by $\mu(a_{21})$, to obtain:
\[\underset{v_{11}+v_{22}\geq-(v_2+v_3+\delta)}{\int_{\underline\lambda,a_{11},a_{22}}}\psi{p}^{6v_1+9v_2+10v_3+7v_4+\delta}p^{v_1+v_2+\min\{0,v_3+v_{22}\}}p^{\gamma}d\mu(\underline\lambda)d\mu(a_{11})\mu(a_{22})=\]\[=\underset{v_{11}+v_{22}\geq-(v_2+v_3+\delta)}{\int_{\underline\lambda,a_{11},a_{22}}}\psi{p}^{7v_1+10v_2+10v_3+7v_4+\delta+\min\{0,v_3+v_{22}\}}p^{\gamma}d\mu(\underline\lambda)d\mu(a_{11})\mu(a_{22}).\]
We denote: $\beta=\min\{0,v_3+v_{22}\}$.
\subsubsection{Computation of constraints on $a_{11}$ and $a_{22}$}
As seen above, we have the constraint 
$m_{4,9}=(a_{11}a_{22}-b_{11})\lambda_2\lambda_3\lambda_4\in\mathbb{Z}_p$, which implies after previous computation that \\
$v_p(a_{11}a_{22}-b_{11})\geq-(v_2+v_3+\min\{v_1,v_4\})$ if \\
$v_p({a_{11})+v_p(a_{22}})\geq-(v_2+v_3+\max\{v_1,v_4\})\implies$ \\
$\implies{v_p}(a_{22})\geq-(v_{11}+v_2+v_3+\max\{v_1,v_4\})$.

Hence, to compute the integral we consider $a_{11}$ as constant, and we integrate by $a_{22}$.
\subsubsubsection{Integration by $a_{22}$}
The constraints $m_{2,6}$ and $m_{4,7}$ together imply that\\
$v(a_{22})\geq\max\{-(v_2+v_3),-(v_3+v_4)\}\geq-(v_3+\min\{v_2,v_4\})$, and it is readily seen that the constraint $m_{5,8}$ is already implied by the constraint $m_{2,6}$.

hence, we have the following constraints:
\begin{enumerate}
    \item $v_{22}\geq-(v_p(v_{11}+v_2+v_3+\max\{v_1,v_4\})$
    \item $v_{22}\geq-(v_3+\min\{v_2,v_4\})$
\end{enumerate}
Hence, $v_{22}\geq-(v_3+\min\{\min\{v_2,v_4\},v_{11}+v_2+\max\{v_1,v_4\}\})=$ \\
$=-(v_3+\min\{v_2,v_4,v_{11}+v_2+\max\{v_1,v_4\}\})$.
We denote $\alpha=\min\{v_2,v_4,v_{11}+v_2+\max\{v_1,v_4\}\}$, and we integrate the expression that is dependent on $a_{22}$ found above, i.e., the measure \\
$\mu(a_{21})=p^{\min\{0,v_3+v_{22}\}}$:

\[\int_{v_{22}\geq-(v_3+\alpha)}p^{\min\{0,v_3+v_{22}\}}.\]

\[
\beta=\begin{cases}
    0,&v_3+v_{22}\geq{0}\iff{v_{22}}\geq{-v_3}\iff\alpha\geq{0}\\
    v_3+v_{22},&v_3+v_{22}<{0}\iff{v_{22}}<{-v_3}\iff\alpha<{0}
\end{cases}
\]

Hence $p^{v_3+(-(v_3+\alpha))}=p^{-\alpha}\leq{p}^{\beta}\leq{p}^0=1$.

We compute all the possible values for $p^{\beta}$ and $\mu(a_{22})$:

\[\begin{matrix}
v_{22}\geq{-v_3}&p^{\beta}=p^{\min\{0,\geq({v_3}-v_3)\}}=p^0=1&\mu(a_{22})=p^{v_3}\\
v_{22}={-(v_3+1)}&p^{\beta}=p^{\min\{0,v_3-(v_3+1)\}}=p^{-1}&\mu(a_{22})=p^{v_3+1}-p^{v_3}=p^{v_3+1}(1-p^{-1})\\
v_{22}={-(v_3+2)}&p^{\beta}=p^{\min\{0,v_3-(v_3+2)\}}=p^{-2}&\mu(a_{22})=p^{v_3+2}-p^{v_3}=p^{v_3+2}(1-p^{-1})\\
\vdots&\vdots&\vdots\\
v_{22}={-(v_3+\alpha)}&p^{\beta}=p^{\min\{0,v_3-(v_3+\alpha)\}}=p^{-\alpha}&\mu(a_{22})=p^{v_3+\alpha}-p^{v_3+\alpha-1}=p^{v_3+\alpha}(1-p^{-1})\\
\end{matrix}\]
All this yields the polynomial: \\
$p^{v_3}+\sum_{i=1}^{\alpha}p^{-i}p^{v_3+i}(1-p^{-1})=p^{3}+\alpha{p}^{v_3}(1-p^{-1})=p^{v_3}(1+\alpha(1-p^{-1}))$, hence we multiply our integral by $\mu(a_{22})$, to obtain:
\[\underset{v_{11}+v_{22}\geq-(v_2+v_3+\delta)}{\int_{\underline\lambda,a_{11}}}\psi{p}^{7v_1+10v_2+10v_3+7v_4+\delta}p^{\gamma}p^{v_3}(1+\alpha(1-p^{-1}))d\mu(\underline\lambda)d\mu(a_{11})=\]\[=\underset{v_{11}+v_{22}\geq-(v_2+v_3+\delta)}{\int_{\underline\lambda,a_{11}}}\psi{p}^{7v_1+10v_2+11v_3+7v_4+\delta}p^{\gamma}(1+\alpha(1-p^{-1}))d\mu(\underline\lambda)d\mu(a_{11}).\]
\subsubsubsection{Integration by $a_{11}$}
Constraint $m_{1,5}$ and $m_{3,6}$ together imply that $v_{11}\geq-(v_2+\min\{v_1,v_3\})$. Considering the constraint $v_{11}+v_{22}\geq-(v_2+v_3+\delta)$, we check that this is already implied by $v_{11}\geq-(v_2+\min\{v_1,v_3\})$. Thus, we integrate by $v_{11}$ under this constraint, where $v_{11}$ depends on $v_1,v_2,v_3,v_4$, to obtain:
\[\underset{v_{11}\geq-(v_2+\min\{v_1,v_3\})}{\int_{\underline\lambda,a_{11}}}\psi{p}^{7v_1+10v_2+11v_3+7v_4+\delta}p^{\gamma}(1+\alpha(1-p^{-1}))d\mu(\underline\lambda)d\mu(a_{11}).\]
We denote $\beta=\min\{v_1,v_3\}$ and $\varphi=1-p^{-1}$.
Since $v_1,v_2,v_3,v_4\geq{0}$, we consider the domain of integration $G^{+}(\mathbb{Q}_p)$ only as the $4$-dimensional orthant of nonnegative integers $(\mathbb{Z}_{\geq{0}})^4$. But the function under the integral, as well as its limits, depend on $3$ choices between pairs of $\lambda$ elements, namely:
\begin{enumerate}
    \item $\delta=\min\{v_1,v_4\}$
    \item $\beta=\min\{v_1,v_3\}$
    \item $\alpha=\min\{v_2,v_4,v_{11}+v_2+\max\{v_1,v_4\}\}=
    \begin{cases}
    \min\{v_2,v_{11}+v_2+\max\{v_1,v_4\}\}&v_2\leq{v_4}\\
    \min\{v_4,v_{11}+v_2+\max\{v_1,v_4\}\}&v_2>{v_4}
    \end{cases}$
\end{enumerate}
